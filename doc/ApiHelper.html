def table_string(response, limit = 5)
  pom = Oj.load(response, mode: :compat)
  pom = [] << pom if pom.class == Hash
  f = StringIO.new
  tp.set :io, f
  tp.set :max_width, 40
  tp pom.take limit
  return f.string
end
